Code Used


# Example of PSD Analysis Code
import mne
import matplotlib.pyplot as plt

# Load the EEG data
raw_data = mne.io.read_raw_brainvision('path/to/your/data.vhdr', preload=True)

# Filter the data (1-50 Hz)
raw_data.filter(1, 50, fir_design='firwin')

# Compute Power Spectral Density (PSD) for Alpha band (8-12 Hz)
psd, freqs = mne.time_frequency.psd_welch(raw_data, fmin=8, fmax=12, n_fft=256)

# Plot the PSD
plt.plot(freqs, psd.mean(axis=0))
plt.title('PSD Alpha Band (8-12 Hz)')
plt.xlabel('Frequency (Hz)')
plt.ylabel('Power (dB)')
plt.show()
